use std::sync::{Arc, RwLock};

/// Thread-safe cache with expiration using RwLock for read-heavy workloads
#[derive(Clone)]
pub struct Cache<V: Clone> {
    inner: Arc<RwLock<CacheInner<V>>>,
    ttl: Duration,
    max_entries: usize,
}

struct CacheInner<V> {
    entries: HashMap<String, CacheEntry<V>>,
    hits: u64,
    misses: u64,
}

struct CacheEntry<V> {
    value: V,
    inserted_at: Instant,
    access_count: u64,
}

impl<V: Clone> Cache<V> {
    pub fn new(ttl: Duration, max_entries: usize) -> Self {
        Self {
            inner: Arc::new(RwLock::new(CacheInner {
                entries: HashMap::new(),
                hits: 0,
                misses: 0,
            })),
            ttl,
            max_entries,
        }
    }

    /// Get a value (read lock - multiple concurrent readers allowed)
    pub fn get(&self, key: &str) -> Option<V> {
        let mut inner = self.inner.write().unwrap();
        if let Some(entry) = inner.entries.get_mut(key) {
            if entry.inserted_at.elapsed() < self.ttl {
                entry.access_count += 1;
                inner.hits += 1;
                return Some(entry.value.clone());
            }
            // Expired - remove it
            inner.entries.remove(key);
        }
        inner.misses += 1;
        None
    }

    /// Set a value (write lock - exclusive access)
    pub fn set(&self, key: String, value: V) {
        let mut inner = self.inner.write().unwrap();
        // Evict if at capacity
        if inner.entries.len() >= self.max_entries && !inner.entries.contains_key(&key) {
            self.evict_lru(&mut inner);
        }
        inner.entries.insert(key, CacheEntry {
            value,
            inserted_at: Instant::now(),
            access_count: 0,
        });
    }

    /// Remove a specific key
    pub fn invalidate(&self, key: &str) -> bool {
        self.inner.write().unwrap().entries.remove(key).is_some()
    }

    /// Get cache statistics
    pub fn stats(&self) -> CacheStats {
        let inner = self.inner.read().unwrap();
        let total = inner.hits + inner.misses;
        CacheStats {
            entries: inner.entries.len(),
            hits: inner.hits,
            misses: inner.misses,
            hit_rate: if total > 0 { inner.hits as f64 / total as f64 } else { 0.0 },
        }
    }

    fn evict_lru(inner: &mut CacheInner<V>) {
        if let Some(lru_key) = inner
            .entries
            .iter()
            .min_by_key(|(_, e)| e.access_count)
            .map(|(k, _)| k.clone())
        {
            inner.entries.remove(&lru_key);
        }
    }
}

    pub hits: u64,
    pub misses: u64,
    pub hit_rate: f64,
}

## Overview

Thread-safe shared state management in Rust using `Arc` (atomic reference counting) with `RwLock` (multiple readers OR one writer) or `Mutex` (exclusive access). This example implements a cache that is safe to share across threads with TTL expiration and LRU eviction.

## Code

```rust
let cache = Cache::new(Duration::from_secs(60), 1000);
let clone = cache.clone(); // Arc::clone - just increments refcount
std::thread::spawn(move || { clone.get("key"); });
```

## Notes

- `RwLock` allows multiple concurrent readers but exclusive write access (better for read-heavy)
- `Arc` enables shared ownership across threads; cloning only increments the atomic counter
- `tokio::sync::Mutex` must be used instead of `std::sync::Mutex` when holding locks across `.await`
- Lock poisoning (std Mutex/RwLock) occurs if a thread panics while holding the lock
- For lock-free alternatives, consider `dashmap` for concurrent HashMaps or atomics for counters
