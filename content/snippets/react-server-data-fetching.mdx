import { cache } from "react";

interface FetchOptions extends RequestInit {
  revalidate?: number | false;
  tags?: string[];
}

async function fetchWithError<T>(url: string, options?: FetchOptions): Promise<T> {
  const { revalidate, tags, ...init } = options ?? {};

  const response = await fetch(url, {
    ...init,
    next: {
      revalidate: revalidate ?? 60,
      tags: tags ?? [],
    },
  });

  if (!response.ok) {
    throw new Error(\`Failed to fetch \${url}: \${response.status} \${response.statusText}\`);
  }

  return response.json();
}

// Cache per-request using React.cache

## Overview

Server Component data fetching patterns that leverage React's streaming architecture. Critical data loads immediately while non-critical sections stream in via Suspense, providing fast initial page loads without client-side waterfalls.

## Code

```typescript
// Parallel fetch initiation
const userPromise = getUser(id);
const postsPromise = getPosts(id);

const user = await userPromise;

return (
  <Suspense fallback={<Skeleton />}>
    <Posts postsPromise={postsPromise} />
  </Suspense>
);
```

## Notes

- `React.cache()` deduplicates identical requests within the same render pass
- Start fetches before `await` to run them in parallel (not sequentially)
- Pass promises to child components to enable streaming without waterfall
- Use `revalidateTag()` in Server Actions for targeted cache invalidation
- `next.revalidate` controls how long data is cached (ISR-style)
- Suspense boundaries define the loading granularity seen by users
