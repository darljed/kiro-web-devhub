export const codeBlocks = {
  main: `#!/usr/bin/env bash
set -euo pipefail

# Colors for output
RED='\\033[0;31m'
GREEN='\\033[0;32m'
YELLOW='\\033[1;33m'
NC='\\033[0m' # No Color

info() { echo -e "\${GREEN}[INFO]\${NC} $1"; }
warn() { echo -e "\${YELLOW}[WARN]\${NC} $1"; }
error() { echo -e "\${RED}[ERROR]\${NC} $1" >&2; }

# Ensure we are in a git repository
if ! git rev-parse --git-dir > /dev/null 2>&1; then
  error "Not a git repository. Run this from your project root."
  exit 1
fi

GIT_HOOKS_DIR="$(git rev-parse --git-dir)/hooks"
PRE_COMMIT_HOOK="$GIT_HOOKS_DIR/pre-commit"

# Detect project type and available tools
detect_tools() {
  local tools=()

  if [[ -f "package.json" ]]; then
    if grep -q '"eslint"' package.json 2>/dev/null; then
      tools+=("eslint")
    fi
    if grep -q '"prettier"' package.json 2>/dev/null; then
      tools+=("prettier")
    fi
    if grep -q '"typescript"' package.json 2>/dev/null; then
      tools+=("tsc")
    fi
    if grep -q '"vitest\\|jest"' package.json 2>/dev/null; then
      tools+=("test")
    fi
  fi

  if [[ -f "pyproject.toml" ]] || [[ -f "setup.py" ]]; then
    if command -v ruff &>/dev/null; then
      tools+=("ruff")
    elif command -v flake8 &>/dev/null; then
      tools+=("flake8")
    fi
    if command -v black &>/dev/null; then
      tools+=("black")
    fi
    if command -v mypy &>/dev/null; then
      tools+=("mypy")
    fi
  fi

  echo "\${tools[@]}"
}

# Generate the pre-commit hook script
generate_hook() {
  local tools
  tools=$(detect_tools)

  cat > "$PRE_COMMIT_HOOK" << 'HOOK_HEADER'
#!/usr/bin/env bash
set -euo pipefail

# Get list of staged files
STAGED_FILES=$(git diff --cached --name-only --diff-filter=ACM)

if [[ -z "$STAGED_FILES" ]]; then
  exit 0
fi

ERRORS=0

run_check() {
  local name="$1"
  shift
  echo "Running $name..."
  if ! "$@"; then
    echo "FAILED: $name"
    ERRORS=$((ERRORS + 1))
  fi
}

HOOK_HEADER

  # Add checks based on detected tools
  for tool in $tools; do
    case "$tool" in
      eslint)
        cat >> "$PRE_COMMIT_HOOK" << 'ESLINT'
# ESLint - check staged JS/TS files
JS_FILES=$(echo "$STAGED_FILES" | grep -E '\\.(js|jsx|ts|tsx)$' || true)
if [[ -n "$JS_FILES" ]]; then
  run_check "ESLint" npx eslint --max-warnings=0 $JS_FILES
fi

ESLINT
        ;;
      prettier)
        cat >> "$PRE_COMMIT_HOOK" << 'PRETTIER'
# Prettier - check formatting
FORMAT_FILES=$(echo "$STAGED_FILES" | grep -E '\\.(js|jsx|ts|tsx|css|md|json)$' || true)
if [[ -n "$FORMAT_FILES" ]]; then
  run_check "Prettier" npx prettier --check $FORMAT_FILES
fi

PRETTIER
        ;;
      tsc)
        cat >> "$PRE_COMMIT_HOOK" << 'TSC'
# TypeScript - type check (runs on full project)
if echo "$STAGED_FILES" | grep -qE '\\.(ts|tsx)$'; then
  run_check "TypeScript" npx tsc --noEmit
fi

TSC
        ;;
      ruff)
        cat >> "$PRE_COMMIT_HOOK" << 'RUFF'
# Ruff - Python linting
PY_FILES=$(echo "$STAGED_FILES" | grep -E '\\.py$' || true)
if [[ -n "$PY_FILES" ]]; then
  run_check "Ruff" ruff check $PY_FILES
fi

RUFF
        ;;
      black)
        cat >> "$PRE_COMMIT_HOOK" << 'BLACK'
# Black - Python formatting
PY_FILES=$(echo "$STAGED_FILES" | grep -E '\\.py$' || true)
if [[ -n "$PY_FILES" ]]; then
  run_check "Black" black --check $PY_FILES
fi

BLACK
        ;;
      mypy)
        cat >> "$PRE_COMMIT_HOOK" << 'MYPY'
# Mypy - Python type checking
if echo "$STAGED_FILES" | grep -qE '\\.py$'; then
  run_check "Mypy" mypy .
fi

MYPY
        ;;
    esac
  done

  # Add exit logic
  cat >> "$PRE_COMMIT_HOOK" << 'HOOK_FOOTER'
if [[ $ERRORS -gt 0 ]]; then
  echo ""
  echo "Pre-commit checks failed ($ERRORS error(s)). Fix issues and try again."
  echo "To skip hooks (emergency only): git commit --no-verify"
  exit 1
fi

echo "All pre-commit checks passed!"
HOOK_FOOTER

  chmod +x "$PRE_COMMIT_HOOK"
}

# Main
info "Detecting project tools..."
TOOLS=$(detect_tools)

if [[ -z "$TOOLS" ]]; then
  warn "No linting/formatting tools detected."
  exit 0
fi

info "Found tools: $TOOLS"
info "Installing pre-commit hook..."

generate_hook

info "Pre-commit hook installed at: $PRE_COMMIT_HOOK"
info "Configured checks: $TOOLS"
echo ""
info "Done! The hook will run automatically on 'git commit'."`,
};

## Overview

This script automatically detects your project's linting and formatting tools, then generates a pre-commit hook tailored to your stack. It supports JavaScript/TypeScript projects (ESLint, Prettier, TypeScript compiler) and Python projects (Ruff, Black, Mypy).

The generated hook only checks staged files (not the entire project) for linting and formatting, keeping commit times fast. Type checking runs on the full project since it needs complete context.

## Notes

- Run this script once per clone: `bash scripts/setup-hooks.sh`
- The hook uses `git diff --cached` to only check files that are staged, not all modified files
- Type checking (`tsc --noEmit`, `mypy`) runs on the full project because partial type checking gives unreliable results
- The `--no-verify` escape hatch is documented in the output for genuine emergencies
- For teams, commit this script to your repo and add a `postinstall` npm script or Makefile target
- Consider using `lint-staged` (npm) or `pre-commit` (Python) framework for more complex setups
- The script is idempotent and can be re-run safely to update the hook when new tools are added
