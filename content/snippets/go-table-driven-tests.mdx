func Divide(a, b float64) (float64, error) {
	if b == 0 {

func TestDivide(t *testing.T) {
	tests := []struct {
		name    string
		a, b    float64
		want    float64
		wantErr bool
	}{
		{name: "positive division", a: 10, b: 2, want: 5},
		{name: "negative result", a: -10, b: 2, want: -5},
		{name: "fractional result", a: 1, b: 3, want: 0.3333333333333333},
		{name: "divide by zero", a: 1, b: 0, wantErr: true},
		{name: "zero numerator", a: 0, b: 5, want: 0},
		{name: "large numbers", a: 1e18, b: 1e9, want: 1e9},
	}

	for _, tt := range tests {
		t.Run(tt.name, func(t *testing.T) {
			t.Parallel() // Run subtests in parallel

			got, err := Divide(tt.a, tt.b)
			if (err != nil) != tt.wantErr {
				t.Errorf("Divide(%v, %v) error = %v, wantErr %v", tt.a, tt.b, err, tt.wantErr)
				return
			}
			if !tt.wantErr && got != tt.want {
				t.Errorf("Divide(%v, %v) = %v, want %v", tt.a, tt.b, got, tt.want)
			}
		})
	}
}

func assertEqual[T comparable](t *testing.T, got, want T) {
	t.Helper()
	if got != want {
		t.Errorf("got %v, want %v", got, want)
	}
}

func assertError(t *testing.T, got error, wantMsg string) {
	t.Helper()
	if got == nil {
		t.Fatal("expected error, got nil")
	}
	if got.Error() != wantMsg {
		t.Errorf("got error %q, want %q", got.Error(), wantMsg)
	}
}

## Overview

The table-driven test pattern is the idiomatic way to write tests in Go. Each test case is a struct in a slice, iterated over with `t.Run()` for named subtests. This provides clear test names in output, parallel execution, and easy addition of new cases.

## Code

```go
tests := []struct {
    name string
    input string
    want  string
}{
    {name: "basic", input: "hello", want: "HELLO"},
}
for _, tt := range tests {
    t.Run(tt.name, func(t *testing.T) {
        t.Parallel()
        got := Transform(tt.input)
        assertEqual(t, got, tt.want)
    })
}
```

## Notes

- `t.Parallel()` runs subtests concurrently for faster test suites
- `t.Helper()` marks functions as test helpers so failures report the correct line
- Named subtests enable running individual cases with `go test -run TestName/case_name`
- Include `setup` functions in test structs for per-case initialization (mocks, fixtures)
- Use `t.Fatalf` for unrecoverable errors and `t.Errorf` for assertion failures
