## Overview

React 19 introduces `useOptimistic` for managing optimistic state updates. This pattern provides instant feedback to users while async operations (like server actions) are in flight. If the operation fails, React automatically rolls back to the previous state on the next render.

The key insight is combining `useOptimistic` with `useTransition` to handle the async boundary cleanly. The optimistic state is only active during the transition; once the server response arrives and the component re-renders with new props, the optimistic state is discarded.

## Code

```typescript
"use client";

import { useOptimistic, useTransition } from "react";

interface Todo {
  id: string;
  text: string;
  completed: boolean;
}

type OptimisticAction =
  | { type: "add"; todo: Todo }
  | { type: "toggle"; id: string }
  | { type: "delete"; id: string };

export function useTodoList(initialTodos: Todo[]) {
  const [optimisticTodos, addOptimistic] = useOptimistic(
    initialTodos,
    (state: Todo[], action: OptimisticAction) => {
      switch (action.type) {
        case "add":
          return [...state, action.todo];
        case "toggle":
          return state.map((t) =>
            t.id === action.id ? { ...t, completed: !t.completed } : t
          );
        case "delete":
          return state.filter((t) => t.id !== action.id);
        default:
          return state;
      }
    }
  );

  const [isPending, startTransition] = useTransition();

  return { optimisticTodos, addOptimistic, isPending, startTransition };
}
```

## Notes

- `useOptimistic` only works in client components (requires `"use client"` directive)
- The optimistic state persists until the parent component re-renders with new data from the server
- Pair with `useTransition` to get `isPending` status for loading indicators
- If your server action throws, React re-renders with the original state, effectively "rolling back" the UI
- For forms, consider combining with the `useActionState` hook for error handling
- The reducer pattern (action types) scales well to complex entity operations
