function validateRequest<T>(schema: z.ZodSchema<T>, data: unknown): {
  success: true; data: T;
} | {
  success: false; errors: { path: string; message: string }[];
} {
  const result = schema.safeParse(data);

  return {
    success: false,

}

## Overview

Runtime schema validation using Zod that generates TypeScript types from schemas, eliminating the need to maintain separate type definitions and validation logic. Schemas serve as the single source of truth for both compile-time types and runtime validation.

## Code

```typescript
const UserSchema = z.object({
  email: z.string().email(),
  name: z.string().min(2).max(100),
  role: z.enum(["admin", "user", "moderator"]),
});

type User = z.infer<typeof UserSchema>;

const result = UserSchema.safeParse(requestBody);
if (!result.success) {
  return { errors: result.error.flatten().fieldErrors };

```

## Notes

- `z.infer` derives TypeScript types from schemas, keeping types and validation in sync
- `safeParse` returns a discriminated union instead of throwing, integrating well with Result patterns
- `.partial()` creates update schemas where all fields are optional
- `.transform()` adds computed fields after validation (like offset from page/limit)
- `z.coerce` handles string-to-number conversion from query parameters automatically
