## Overview

A lightweight custom hook that delays updating a value until a specified period of inactivity. This is essential for preventing excessive API calls during user input, such as search-as-you-type functionality.

## Code

```typescript
import { useState, useEffect } from "react";

export function useDebounce<T>(value: T, delay: number): T {
  const [debouncedValue, setDebouncedValue] = useState<T>(value);

  useEffect(() => {
    const timer = setTimeout(() => setDebouncedValue(value), delay);
    return () => clearTimeout(timer);
  }, [value, delay]);

  return debouncedValue;
}
```

## Usage

```tsx
import { useState } from "react";
import { useDebounce } from "@/hooks/useDebounce";

export function SearchInput() {
  const [query, setQuery] = useState("");
  const debouncedQuery = useDebounce(query, 300);

  useEffect(() => {
    if (debouncedQuery) {
      // Fetch search results
      fetchResults(debouncedQuery);
    }
  }, [debouncedQuery]);

  return (
    <input
      value={query}
      onChange={(e) => setQuery(e.target.value)}
      placeholder="Search..."
    />
  );
}
```

## Notes

- The delay parameter is in milliseconds (300-500ms is common for search inputs)
- The hook cleans up timers on unmount to prevent memory leaks
- Works with any value type thanks to TypeScript generics
- For more complex scenarios, consider `useDeferredValue` from React 19
