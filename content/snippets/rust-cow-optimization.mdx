use std::collections::HashMap;

/// Normalize a URL path - only allocates if modification is needed
pub fn normalize_path(path: &str) -> Cow<'_, str> {
    // Fast path: already normalized
    if !path.contains("//")
        && !path.ends_with('/')
        && !path.contains("/../")
        && !path.contains("/./")
    {
        return Cow::Borrowed(path);
    }

    // Slow path: needs modification - allocate only here
    let mut normalized = String::with_capacity(path.len());
    let mut prev_slash = false;

    for ch in path.chars() {
        if ch == '/' {
            if !prev_slash {
                normalized.push(ch);
            }
            prev_slash = true;
        } else {
            normalized.push(ch);
            prev_slash = false;
        }
    }

    // Remove trailing slash (except for root)
    if normalized.len() > 1 && normalized.ends_with('/') {
        normalized.pop();
    }

    Cow::Owned(normalized)
}

/// Template engine that avoids allocation when no variables are present
pub fn render_template<'a>(
    template: &'a str,
    variables: &HashMap<&str, &str>,
) -> Cow<'a, str> {
    if !template.contains("{{") {
        return Cow::Borrowed(template);
    }

    let mut result = String::with_capacity(template.len());
    let mut chars = template.chars().peekable();

    while let Some(ch) = chars.next() {
        if ch == '{' && chars.peek() == Some(&'{') {
            chars.next(); // consume second {
            let mut var_name = String::new();
            while let Some(&c) = chars.peek() {
                if c == '}' {
                    chars.next();
                    if chars.peek() == Some(&'}') {
                        chars.next();
                    }
                    break;
                }
                var_name.push(c);
                chars.next();
            }
            let var_name = var_name.trim();
            if let Some(value) = variables.get(var_name) {
                result.push_str(value);
            } else {
                result.push_str("{{");
                result.push_str(var_name);
                result.push_str("}}");
            }
        } else {
            result.push(ch);
        }
    }

    Cow::Owned(result)
}

## Overview

`Cow` (Clone on Write) enables functions to return either a borrowed reference or an owned value, avoiding allocation in the common case where no modification is needed. This is particularly effective for string processing where most inputs require no changes.

## Code

```rust
pub fn process(input: &str) -> Cow<'_, str> {
    if needs_modification(input) {
        Cow::Owned(modify(input))  // Allocates only when needed
    } else {
        Cow::Borrowed(input)  // Zero-cost: just a reference
    }
}
```

## Notes

- `Cow::Borrowed` is a thin wrapper around a reference (no heap allocation)
- `Cow::Owned` holds an owned value (allocated on heap)
- Both variants implement `Deref` to the inner type, so they can be used as `&str` transparently
- This pattern shines in parsers, validators, and normalizers where most inputs pass through unchanged
- `into_owned()` converts to the owned variant (cloning if borrowed) when ownership is required
