/// Application-level error type that unifies all error sources
#[derive(Error, Debug)]
pub enum AppError {
    #[error("Database error: {0}")]
    Database(#[from] DatabaseError),

    #[error("Validation error: {field} - {message}")]
    Validation { field: String, message: String },

    #[error("User '{id}' not found")]
    NotFound { id: String, resource: String },

    Unauthorized(String),

    #[error("Rate limited: retry after {retry_after_secs}s")]
    RateLimited { retry_after_secs: u64 },

    #[error("External service error: {service}")]
    ExternalService {
        service: String,
        #[source]
        source: reqwest::Error,
    },

    #[error(transparent)]
    Internal(#[from] anyhow::Error),
}

/// Database-specific errors
#[derive(Error, Debug)]
pub enum DatabaseError {
    #[error("Connection failed: {0}")]
    Connection(#[source] io::Error),

    #[error("Query timeout after {timeout_ms}ms")]
    Timeout { timeout_ms: u64 },

    #[error("Unique constraint violation on {column}")]
    UniqueViolation { column: String },

    #[error("Migration failed: {0}")]
    Migration(String),
}

    pub fn status_code(&self) -> u16 {
        match self {
            Self::NotFound { .. } => 404,
            Self::Validation { .. } => 400,
            Self::Unauthorized(_) => 401,
            Self::RateLimited { .. } => 429,
            Self::Database(DatabaseError::UniqueViolation { .. }) => 409,
            _ => 500,
        }
    }

    /// Create a user-safe error message (hide internals)
    pub fn user_message(&self) -> &str {
        match self {
            Self::NotFound { .. } => "Resource not found",
            Self::Validation { .. } => "Invalid request",
            Self::Unauthorized(_) => "Authentication required",
            Self::RateLimited { .. } => "Too many requests",
            Self::Database(DatabaseError::UniqueViolation { .. }) => "Resource already exists",
            _ => "Internal server error",
        }
    }

    /// Whether the operation should be retried
    pub fn is_retryable(&self) -> bool {
        matches!(
            self,
            Self::Database(DatabaseError::Connection(_) | DatabaseError::Timeout { .. })
                | Self::RateLimited { .. }
                | Self::ExternalService { .. }
        )
    }
}

pub type AppResult<T> = Result<T, AppError>;

## Overview

Structured error handling using `thiserror` for automatic `Error` trait derivation. Each error variant carries contextual data, and the `#[from]` attribute enables automatic conversions. Error types expose methods for HTTP mapping, user-safe messages, and retry decisions.

## Code

```rust
#[derive(Error, Debug)]
pub enum AppError {
    #[error("User '{id}' not found")]
    NotFound { id: String },

    #[error("Database error: {0}")]
    Database(#[from] DatabaseError),
}
```

## Notes

- `#[from]` generates `From` implementations enabling the `?` operator for automatic conversion
- `#[source]` preserves the error chain for `.source()` traversal without creating `From` impls
- `#[error(transparent)]` delegates both Display and source to the inner error
- `status_code()` and `user_message()` centralize HTTP response mapping
- `is_retryable()` helps callers decide whether to retry failed operations
