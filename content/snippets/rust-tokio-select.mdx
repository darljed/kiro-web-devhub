pub enum Message {
    Process { id: String, payload: Vec<u8>, reply: oneshot::Sender<Result<(), String>> },
    HealthCheck { reply: oneshot::Sender<HealthStatus> },

#[derive(Debug)]
pub struct HealthStatus {

    pub errors: u64,

impl MessageProcessor {
    pub fn new(rx: mpsc::Receiver<Message>, shutdown: broadcast::Receiver<()>) -> Self {
        Self {
            rx,
            shutdown,
            state: ProcessorState {
                processed: 0,
                errors: 0,
                last_message_at: None,
            },

    pub async fn run(&mut self) {
        let mut cleanup_interval = time::interval(Duration::from_secs(60));
        let mut idle_timeout = time::interval(Duration::from_secs(300));

        loop {
            tokio::select! {
                // Priority: shutdown signal (biased ensures this is checked first)
                biased;

                _ = self.shutdown.recv() => {

                        }

                }
            }
        }
    }

        ).await;

            Ok(Ok(())) => Ok(()),
            Ok(Err(e)) => Err(e),
            Err(_) => Err(format!("Processing {} timed out", id)),

        while let Ok(msg) = self.rx.try_recv() {

        }

/// Request-response pattern with timeout
pub async fn request_with_timeout<T>(
    tx: &mpsc::Sender<Message>,
    build_msg: impl FnOnce(oneshot::Sender<T>) -> Message,
    timeout: Duration,
) -> Result<T, RequestError> {
    let (reply_tx, reply_rx) = oneshot::channel();
    let msg = build_msg(reply_tx);

    tx.send(msg).await.map_err(|_| RequestError::ChannelClosed)?;

    tokio::select! {
        result = reply_rx => {
            result.map_err(|_| RequestError::Cancelled)
        }
        _ = tokio::time::sleep(timeout) => {
            Err(RequestError::Timeout)
        }
    }
}

#[derive(Debug)]
pub enum RequestError {
    Timeout,
    Cancelled,
    ChannelClosed,

## Overview

`tokio::select!` enables waiting on multiple async operations simultaneously, processing whichever completes first. This pattern is essential for building responsive async services that handle messages, timers, and shutdown signals concurrently.

## Code

```rust
tokio::select! {
    biased;
    _ = shutdown.recv() => break,
    Some(msg) = rx.recv() => handle(msg).await,
    _ = interval.tick() => cleanup().await,
}
```

## Notes

- `biased;` ensures branches are checked in order (useful for prioritizing shutdown signals)
- Without `biased`, select randomly picks among ready branches to prevent starvation
- `oneshot` channels provide request-response patterns with backpressure
- `broadcast` channels are ideal for shutdown signals (multiple receivers)
- Always handle the case where channels close (`None` from `.recv()`)
