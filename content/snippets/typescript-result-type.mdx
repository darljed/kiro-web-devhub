// Wrap a throwing function into a Result
export function tryCatch<T, E = Error>(fn: () => T): Result<T, E> {
  try {
    return Ok(fn());
  } catch (error) {
    return Err(error as E);
  }
}

// Wrap an async throwing function into a Result
export async function tryCatchAsync<T, E = Error>(
  fn: () => Promise<T>
): Promise<Result<T, E>> {
  try {
    const value = await fn();
    return Ok(value);
  } catch (error) {
    return Err(error as E);
  }
}

// Utility: map over the success value
export function mapResult<T, U, E>(
  result: Result<T, E>,
  fn: (value: T) => U
): Result<U, E> {
  if (result.ok) {
    return Ok(fn(result.value));
  }
  return result;
}

export function flatMapResult<T, U, E>(
  result: Result<T, E>,
  fn: (value: T) => Result<U, E>
): Result<U, E> {
  if (result.ok) {
    return fn(result.value);
  }
  return result;
}

// Utility: unwrap with default
export function unwrapOr<T, E>(result: Result<T, E>, defaultValue: T): T {
  return result.ok ? result.value : defaultValue;

export function collectResults<T, E>(results: Result<T, E>[]): Result<T[], E> {

  return Ok(values);

## Overview

A Result type that encodes success and failure in the type system, forcing callers to explicitly handle both cases. This eliminates unhandled exceptions and makes error paths visible in function signatures, inspired by Rust's `Result<T, E>`.

## Code

```typescript
type Result<T, E = Error> =
  | { ok: true; value: T }
  | { ok: false; error: E };

export function Ok<T>(value: T): Result<T, never> {
  return { ok: true, value };
}

export function Err<E>(error: E): Result<never, E> {
  return { ok: false, error };
}
```

## Notes

- Unlike exceptions, Result types make error handling explicit and visible in function signatures
- Use discriminated union domain errors (`type` field) for exhaustive pattern matching
- `tryCatch` and `tryCatchAsync` wrap existing throwing code into the Result pattern
- `collectResults` is useful when running multiple operations that may each fail independently
- This pattern works well with the `assertNever` exhaustive check for error handling
