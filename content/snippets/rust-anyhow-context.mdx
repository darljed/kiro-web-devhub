/// Load and parse a configuration file with rich error context
pub fn load_config(path: &Path) -> Result<AppConfig> {
    let content = fs::read_to_string(path)
        .with_context(|| format!("Failed to read config file: {}", path.display()))?;

    let config: AppConfig = toml::from_str(&content)
        .with_context(|| format!("Failed to parse config file: {}", path.display()))?;

    validate_config(&config)
        .with_context(|| "Config validation failed")?;

    Ok(config)
}

/// Validate configuration with bail! for early returns
fn validate_config(config: &AppConfig) -> Result<()> {
    ensure!(
        !config.database_url.is_empty(),
        "database_url must not be empty"
    );

    ensure!(
        config.port > 0 && config.port < 65536,
        "port must be between 1 and 65535, got {}",
        config.port
    );

    if config.workers == 0 {
        bail!("workers must be at least 1");
    }

    // Validate paths exist
    if let Some(ref cert_path) = config.tls_cert {
        ensure!(
            Path::new(cert_path).exists(),
            "TLS certificate not found: {}",
            cert_path
        );
    }

    Ok(())
}

/// Downcast to inspect specific error types
pub fn handle_startup_error(error: &anyhow::Error) {
    // Walk the error chain
    eprintln!("Error: {}", error);
    for (i, cause) in error.chain().enumerate().skip(1) {
        eprintln!("  Caused by [{i}]: {cause}");
    }

    // Check for specific error types
    if let Some(io_err) = error.downcast_ref::<std::io::Error>() {
        match io_err.kind() {
            std::io::ErrorKind::PermissionDenied => {
                eprintln!("Hint: Check file permissions or run with elevated privileges");
            }
            std::io::ErrorKind::AddrInUse => {
                eprintln!("Hint: Another process is using this port. Try a different port.");
            }
            _ => {}
        }
    }

    // Check for our domain errors
    if error.downcast_ref::<DatabaseError>().is_some() {
        eprintln!("Hint: Ensure the database is running and accessible");
    }
}

## Overview

`anyhow` provides application-level error handling with context chains that show what each layer was doing when the error occurred. Unlike `thiserror` (for library error types), `anyhow` is designed for applications where you want rich debugging information without defining error types for every failure mode.

## Notes

- `with_context()` wraps errors with additional context forming a chain (newest first)
- `bail!` is shorthand for `return Err(anyhow!(...))` with format string support
- `ensure!` is shorthand for `if !condition { bail!(...) }`
- `error.chain()` iterates through all layers for display
- `downcast_ref()` enables inspecting the original error type for recovery decisions
- Use `thiserror` for library errors (structured) and `anyhow` for applications (convenience)
