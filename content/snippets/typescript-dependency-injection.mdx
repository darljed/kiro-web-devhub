type Scope = "singleton" | "transient";

interface Registration<T> {
  factory: Factory<T>;
  scope: Scope;
  instance?: T;
}

export class Container {
  private registrations = new Map<string, Registration<unknown>>();

  register<T>(token: string, factory: Factory<T>, scope: Scope = "singleton"): this {
    this.registrations.set(token, { factory, scope });
    return this;
  }

  resolve<T>(token: string): T {
    const registration = this.registrations.get(token);
    if (!registration) {
      throw new Error(\`No registration found for token: \${token}\`);
    }

    if (registration.scope === "singleton") {
      if (!registration.instance) {
        registration.instance = registration.factory();
      }
      return registration.instance as T;
    }

    return registration.factory() as T;
  }

  has(token: string): boolean {
    return this.registrations.has(token);
  }

  reset(): void {
    this.registrations.clear();
  }
}

## Overview

A lightweight dependency injection container that manages service lifetimes and dependencies without decorators or reflection. Services are registered with factory functions and resolved lazily, making it easy to swap implementations for testing.

## Code

```typescript
export class Container {
  private registrations = new Map<string, Registration<unknown>>();

  register<T>(token: string, factory: Factory<T>, scope: Scope = "singleton"): this {
    this.registrations.set(token, { factory, scope });
    return this;
  }

  resolve<T>(token: string): T {
    const registration = this.registrations.get(token);
    if (registration.scope === "singleton") {
      if (!registration.instance) {
        registration.instance = registration.factory();
      }
      return registration.instance as T;
    }
    return registration.factory() as T;
  }
}
```

## Notes

- Singleton scope creates one instance shared across all resolutions
- Transient scope creates a new instance on every `resolve()` call
- The composition root (`createContainer`) is the single place where all dependencies are wired
- No decorators or `reflect-metadata` needed, keeping the runtime overhead minimal
- For larger applications, consider hierarchical containers with parent/child scopes
